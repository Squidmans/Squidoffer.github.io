#include<stdio.h>

#include<stdlib.h>

#include<time.h>
int main(void)
{
int guess=0;
int count=4;
int chosen=0;
int limit=20;
srand(time(NULL));
chosen=1+rand()%limit;

printf("This is a guessing game.\nYou have to choose a number between 1 to 20 to join in.");
for(;count>0;--count)
{
printf("\nyou have %d tr%s left",count,count==1?"y":"ies");
printf("\nNow you can enter a guess:");
scanf("%d",&guess);
if((guess>20)||(guess<1))
guess=21;
if(guess==chosen)
guess=22;
switch(guess)
{
    case 22:
        printf("\nCongratulations,you win the first prize.");
        goto there;
    case 21:
        printf("\n you are not putting a right number.");
        break;
    default:
        printf("\nthe number is %s than this.",guess>chosen?"more":"less");
        break;
}
}
printf("\nyou have had four tries and failed.the number is %d.",chosen);
there:return 0;

}
